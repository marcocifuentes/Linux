## Instalando servidor de arquivos SAMBA
sudo aot install samba

##definindo diretorio para compartilhamento
mkdir /public

##alterando permissoes total (rwxrwxrwx) p/ todos os usuarios
chmod 777 public/

##alterando configuracoes no smb.conf
nano /etc/samba/smb.conf
# inserir, no final do arquivo, parametros p/ compartilhar diretorio
[public]
path = /public
writable = yes
guest ok = yes
guest only = yes

##reiniciar o servico do servidor SAMBA em segundo plano (daemon)
systemctl restart smbd

##verificar status do servico (se esta ouvindo requisicoes)
systemctl status smbd

##definir execucao automatica do servico (caso reinicie o servidor)
systemctl anable smbd
